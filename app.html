// --- ÜST TARAFIN (HTML/CSS) AYNI KALDIĞINI VARSAYIYORUM, SADECE SCRIPT'İ GÜNCELLE --- let ax = 0, ay = 0, az = 0; let sensorBuffer = []; const windowSize = 50; const sınıflar = ["kosma", "merdiven", "oturma", "yurume", "ziplama"]; let sureler = { "oturma": 0, "yurume": 0, "kosma": 0, "ziplama": 0, "merdiven": 0 }; // EKRANA LOG YAZDIRMA FONKSİYONU (DEDEKTİFİMİZ BU) function logYaz(mesaj) { const logBox = document.getElementById("durum"); document.getElementById("analiz-ekrani").style.display = "flex"; baslat(); } else { logYaz("HATA: İzin reddedildi."); } }) .catch(err => logYaz("HATA: " + err)); } else { document.getElementById("tanitim-ekrani").style.display = "none"; document.getElementById("analiz-ekrani").style.display = "flex"; baslat(); } } function baslat() { logYaz("Sensör dinleniyor..."); // Model yüklendi mi kontrol et (25MB olduğu için geç yüklenebilir) if (typeof score === 'undefined') { logYaz("HATA: Model dosyası (model.js) henüz yüklenemedi veya çok büyük olduğu için tarayıcı tarafından durduruldu!"); return; } window.addEventListener('devicemotion', (event) => { let acc = event.accelerationIncludingGravity; if (acc && acc.x !== null) { ax = acc.x; ay = acc.y; az = acc.z; sensorBuffer.push([ax, ay, az]); if (sensorBuffer.length >= windowSize) { tahminEt(sensorBuffer); sensorBuffer = []; } } else { logYaz("Sensöre bağlandı ama veri gelmiyor..."); } }, true); } function tahminEt(data) { try { // Özellikleri hesapla let sumX = 0, sumY = 0, sumZ = 0; let mags = []; for (let i = 0; i < data.length; i++) { sumX += data[i][0]; sumY += data[i][1]; sumZ += data[i][2]; mags.push(Math.sqrt(data[i][0]**2 + data[i][1]**2 + data[i][2]**2)); } let inputFeatures = [ sumX / data.length, sumY / data.length, sumZ / data.length, 0, 0, 0, Math.max(...data.map(d => d[0])), Math.max(...data.map(d => d[1])), Math.max(...data.map(d => d[2])), Math.min(...data.map(d => d[0])), Math.min(...data.map(d => d[1])), Math.min(...data.map(d => d[2])), 0, 0, 0, mags.reduce((a, b) => a + b, 0) / mags.length, 0 ]; // TAHMİNİ ÇALIŞTIR let sonucIndex = score(inputFeatures); // model.js içindeki fonksiyon let tahminEdilen = sınıflar[sonucIndex]; document.getElementById("tahmin").innerText = tahminEdilen; logYaz("Analiz aktif!"); // Süreleri güncelle (kodun devamı aynı...) sureler[tahminEdilen]++; document.getElementById(`time-${tahminEdilen}`).innerText = sureler[tahminEdilen] + " sn"; } catch (e) { logYaz("TAHMİN HATASI: " + e.message); } }
